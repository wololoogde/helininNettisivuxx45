<script>
    const chat = document.getElementById('chat');
    const userInput = document.getElementById('userInput');

    function sendMessage() {
        const message = userInput.value.trim();
        if(message === '') return;

        // Näytä käyttäjän viesti
        const userMsg = document.createElement('div');
        userMsg.className = 'message user';
        userMsg.textContent = "Sinä: " + message;
        chat.appendChild(userMsg);

        // Tyhjennä kenttä
        userInput.value = '';

        // Botin vastaus
        const botMsg = document.createElement('div');
        botMsg.className = 'message bot';
        botMsg.textContent = "Assistentti: " + generateBotResponse(message);
        chat.appendChild(botMsg);

        // Scrollaa automaattisesti alas
        chat.scrollTop = chat.scrollHeight;
    }

    function generateBotResponse(input) {
        // Yksinkertainen vastauslogiikka
        input = input.toLowerCase();
        if(input.includes("hei")) return "Hei! Miten voin auttaa?";
        if(input.includes("mitä voit tehdä")) return "Voin keskustella kanssasi ja vastata yksinkertaisiin kysymyksiin.";
        if(input.includes("kuka olet")) return "Olen Helinin nettisivun tekoälyassistentti!";
        if(input.includes("kiitos")) return "Ole hyvä!";
        return "Hmmm… en ole varma miten vastata siihen. Kysy jotain muuta!";
    }

    // Salli Enter-näppäimen käyttö
    userInput.addEventListener("keypress", function(event) {
        if(event.key === "Enter") sendMessage();
    });
</script>
